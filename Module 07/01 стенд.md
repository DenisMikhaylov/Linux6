node1# cat /etc/bind/corp.ru
```

```

Сервис Keepalived 
```
node1# cat /etc/keepalived/keepalived.conf
```
```
global_defs {
router_id uMASTER
enable_script_security
script_user root
}
vrrp_script myhealth {
    script "/bin/nc -z -w 2 127.0.0.1 53"
    interval 10
    user nobody
}
vrrp_instance VI_1 {

    state MASTER
    interface br0
    priority 90
    advert_int 1
    nopreempt

        authentication {
        auth_type PASS
        auth_pass secret
        }

    virtual_router_id 1
    virtual_ipaddress {
        192.168.10.254
    }
    virtual_routes {
        0.0.0.0/0 via 192.168.123.2 dev eth0
    }
    track_script {
        myhealth
    }
}

```
